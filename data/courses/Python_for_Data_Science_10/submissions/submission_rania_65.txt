def calculate_average(numbers):
    if not numbers:
        return 0
    try:
        return sum(numbers) / len(numbers)
    except TypeError:
        return "Error: List contains non-numeric values"

# Test cases
print(calculate_average([10, 20, 30, 40])) # Expected: 25.0
print(calculate_average([10, 'a', 30]))    # Expected: Error message
